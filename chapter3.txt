<!DOCTYPE html>
<html>
<head>
  <title>JS Timer</title>
</head>
<body>

<h1 id="display">0</h1>

<button onclick="startTimer()">Start</button>
<button onclick="stopTimer()">Stop</button>
<button onclick="resetTimer()">Reset</button>

<script>
  let time = 0;
  let interval = null;

  function startTimer() {
    if (interval) return; // prevent multiple timers
    interval = setInterval(() => {
      time++;
      document.getElementById("display").innerText = time;
    }, 1000);
  }

  function stopTimer() {
    clearInterval(interval);
    interval = null;
  }

  function resetTimer() {
    stopTimer();
    time = 0;
    document.getElementById("display").innerText = time;
  }
</script>

</body>
</html>
